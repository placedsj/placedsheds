frontend/js/lunai.js (AI Chat Interface)

class LunAI {
    constructor() {
        this.messages = [];
        this.currentStep = 0;
        this.design = {
            size: null,
            style: null,
            siding: null,
            roof: null,
            addons: []
        };
    }

    async start() {
        this.addMessage("Hi! I'm LunAI. Let's design your perfect shed! What size are you thinking?", "ai");
        this.showOptions([
            "8x10 ($3,500)",
            "10x12 ($4,200)",
            "12x16 ($6,500)",
            "12x20 ($8,000)",
            "Custom size"
        ]);
    }

    addMessage(text, sender) {
        const chatContainer = document.getElementById('lunai-chat');
        const messageDiv = document.createElement('div');
        messageDiv.className = `message message-${sender}`;
        messageDiv.textContent = text;
        chatContainer.appendChild(messageDiv);
        chatContainer.scrollTop = chatContainer.scrollHeight;
    }

    showOptions(options) {
        const chatContainer = document.getElementById('lunai-chat');
        const optionsDiv = document.createElement('div');
        optionsDiv.className = 'options-container';
        
        options.forEach(option => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.textContent = option;
            btn.onclick = () => this.handleSelection(option);
            optionsDiv.appendChild(btn);
        });
        
        chatContainer.appendChild(optionsDiv);
    }

    async handleSelection(selection) {
        this.addMessage(selection, "user");
        
        // Process selection based on current step
        switch(this.currentStep) {
            case 0:
                this.design.size = selection;
                this.currentStep = 1;
                this.addMessage("Great choice! What style?", "ai");
                this.showOptions(["A-Frame", "Lofted Barn", "Modern Saltbox", "Quaker"]);
                break;
            case 1:
                this.design.style = selection;
                this.currentStep = 2;
                this.addMessage("Perfect! What siding?", "ai");
                this.showOptions(["Smart Panel T1-11 (Included)", "Vinyl Dutchlap (+$800)", "Cedar (+$1,200)"]);
                break;
            case 2:
                this.design.siding = selection;
                this.currentStep = 3;
                this.addMessage("Nice! Roof type?", "ai");
                this.showOptions(["Asphalt Shingles (Included)", "Metal Standing Seam (+$400)"]);
                break;
            case 3:
                this.design.roof = selection;
                this.currentStep = 4;
                this.addMessage("Almost done! Any add-ons?", "ai");
                this.showOptions(["Extra Windows (+$150)", "Skylights (+$300)", "Electrical (+$800)", "No thanks, calculate price"]);
                break;
            case 4:
                if (selection !== "No thanks, calculate price") {
                    this.design.addons.push(selection);
                    this.showOptions(["Extra Windows (+$150)", "Skylights (+$300)", "Electrical (+$800)", "Done, calculate price"]);
                } else {
                    await this.calculatePrice();
                }
                break;
        }
    }

    async calculatePrice() {
        this.addMessage("Calculating your custom shed price...", "ai");
        
        // Call backend API
        const response = await fetch('/api/calculate-price', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(this.design)
        });
        
        const data = await response.json();
        
        document.getElementById('price-display').innerHTML = `
            <h2>Your Custom Shed</h2>
            <p>Total: $${data.total.toFixed(2)} CAD</p>
            <p>Or $${data.monthly_payment}/month (36 months, 0% interest)</p>
            <button class="btn-primary" onclick="requestQuote()">Request Quote & Build Doc</button>
        `;
    }
}

// Initialize on page load
let lunai;
function openDesigner() {
    document.getElementById('designer-container').classList.remove('hidden');
    lunai = new LunAI();
    lunai.start();
}

# Models
class ShedDesign(BaseModel):
    size: str
    style: str
    siding: str
    roof: str
    addons: List[str] = []

class PricingResponse(BaseModel):
    materials: float
    labor: float
    addons_cost: float
    subtotal: float
    tax: float
    total: float
    monthly_payment: float

# Routes
@app.get("/")
async def serve_homepage():
    from fastapi.responses import FileResponse
    return FileResponse("../frontend/index.html")

@app.post("/api/calculate-price", response_model=PricingResponse)
async def calculate_shed_price(design: ShedDesign):
    """Calculate shed price based on design"""
    
    # Base pricing
    base_prices = {
        "8x10 ($3,500)": 1580,
        "10x12 ($4,200)": 2150,
        "12x16 ($6,500)": 2950,
        "12x20 ($8,000)": 3800
    }
    
    materials = base_prices.get(design.size, 2150)
    
    # Add siding upcharge
    if "Vinyl" in design.siding:
        materials += 800
    elif "Cedar" in design.siding:
        materials += 1200
    
    # Add roof upcharge
    if "Metal" in design.roof:
        materials += 400
    
    # Add add-ons
    addon_costs = {
        "Extra Windows (+$150)": 150,
        "Skylights (+$300)": 300,
        "Electrical (+$800)": 800
    }
    
    addons_cost = sum(addon_costs.get(addon, 0) for addon in design.addons)
    
    # Labor (flat $2K)
    labor = 2000
    
    # Calculate total
    subtotal = materials + labor + addons_cost
    tax = subtotal * 0.15  # HST
    total = subtotal + tax
    monthly_payment = round(total / 36, 2)
    
    return PricingResponse(
        materials=materials,
        labor=labor,
        addons_cost=addons_cost,
        subtotal=subtotal,
        tax=tax,
        total=total,
        monthly_payment=monthly_payment
    )
